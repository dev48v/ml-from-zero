📐 Support Vector Machines — The Widest-Margin Classifier

Click to add points and watch the widest "street" appear

SVM finds the line with the widest margin between the two classes. Only the closest points — the support vectors (circled) — hold the street in place. Everything else could move freely and the line wouldn't budge.

Low C = wide, forgiving street (tolerates misclassified points). High C = tight street that tries hard to classify every point correctly.

const data = [

    The model: a line w·x + b
    
        
    

    let w = [0, 0], b = 0;                       // start flat
const score   = x => w[0]*x[0] + w[1]*x[1] + b;   // signed distance-ish
const predict = x => (score(x) >= 0 ? +1 : -1);

    The hinge loss: penalise points inside the margin
    
        
    

    // loss = ½‖w‖²  +  C · Σ max(0, 1 − y·(w·x + b))
//          ↑ wide margin            ↑ pay only when a point
//                                     is wrong OR inside the street
function loss(C) {
  const reg = 0.5 * (w[0]*w[0] + w[1]*w[1]);
  const hinge = data.reduce((s, p) =>
    s + Math.max(0, 1 - p.y * score(p.x)), 0);
  return reg + C * hinge;
}

  4. 
        
    4
    Gradient descent on w and b
    
        
    

    function step(C, lr) {
  let dw = [w[0], w[1]], db = 0;             // gradient of ½‖w‖² is w
  for (const p of data) {
    if (p.y * score(p.x) < 1) {              // point violates the margin
      dw[0] -= C * p.y * p.x[0];
      dw[1] -= C * p.y * p.x[1];
      db    -= C * p.y;
    }                                        // else: only the ‖w‖² term pulls
  }
  w[0] -= lr * dw[0];  w[1] -= lr * dw[1];  b -= lr * db;
}
for (let i = 0; i < 2000; i++) step(1.0, 0.01);   // train

const supportVectors = data.filter(p =>

const marginWidth = 2 / Math.hypot(w[0], w[1]);   // street width = 2/‖w‖

    Plot the boundary and the two margin lines
    
        
    

    // decision boundary:  w·x + b = 0
// margin lines:        w·x + b = +1   and   w·x + b = −1
// solve for x2 given x1:  x2 = -(w0·x1 + b - level) / w1
function lineY(x1, level) { return -(w[0]*x1 + b - level) / w[1]; }
drawLine(lineY(0, 0),  lineY(1, 0),  "solid");    // boundary
drawLine(lineY(0, +1), lineY(1, +1), "dashed");   // margin edge
drawLine(lineY(0, -1), lineY(1, -1), "dashed");   // margin edge

    Go non-linear: the kernel trick
    
        
    

    // can't split a ring with a line? swap dot-products for a KERNEL.
// RBF kernel = "how close are two points" (1 if identical, →0 if far)
const rbf = (a, b, g = 8) =>
  Math.exp(-g * ((a[0]-b[0])**2 + (a[1]-b[1])**2));
// the math stays 2D, but distances are measured so a curved
// boundary in the original space becomes a flat one in kernel space.
