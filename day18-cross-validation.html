🔨 Build it yourself

    
k-fold cross-validation from scratch: shuffle the data and slice it into k folds, loop so each fold takes its turn as validation while the rest train, score every fold and report the mean ± std, then grid-search a hyperparameter using CV to pick the best setting — all while keeping a final test set untouched. The exact loop scikit-learn's cross_val_score and KFold run for you, now demystified.

    Shuffle, then split into k folds
    
        
    

    # shuffle so the folds aren't accidentally ordered (e.g. sorted by
# label or by time). then chop the shuffled indices into k roughly
# equal chunks — those chunks are your folds.
import numpy as np
def make_folds(n, k, seed=0):
    idx = np.random.default_rng(seed).permutation(n)   # shuffle
    return np.array_split(idx, k)                       # k chunks
folds = make_folds(len(X), k=5)
# each fold is a set of row-indices that will be the validation set once.

    3
    Fit & score inside the fold (no leakage)
    
        
    

    # CRITICAL: fit any preprocessing (scaler, imputer, encoder) on the
# TRAINING fold only, then apply it to the validation fold. fitting a
# scaler on the whole dataset leaks validation stats into training.
def fit_and_score(X_tr, y_tr, X_va, y_va):
    mu, sd = X_tr.mean(0), X_tr.std(0) + 1e-9   # stats from TRAIN only
    Xtr, Xva = (X_tr - mu) / sd, (X_va - mu) / sd
    w = np.linalg.lstsq(Xtr, y_tr, rcond=None)[0]   # train the model
    pred = Xva @ w
    return ((pred - y_va) ** 2).mean()              # validation MSE

    Average the folds: mean ± std
    
        
    

    # collect the k validation scores, then summarise them.
# the MEAN is your honest performance estimate; the STD tells you how
# sensitive the model is to which data it saw (stability).
def cross_validate(X, y, k=5):
    folds = make_folds(len(X), k)
    scores = []
    for i in range(k):
        val = folds[i]
        tr  = np.concatenate([folds[j] for j in range(k) if j != i])
        scores.append(fit_and_score(X[tr], y[tr], X[val], y[val]))
    scores = np.array(scores)
    return scores.mean(), scores.std()      # estimate ± spread

    In the real world: scikit-learn
    
        
    

    from sklearn.model_selection import cross_val_score, KFold, GridSearchCV
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import Ridge

pipe = make_pipeline(StandardScaler(), Ridge())   # scaler fit per-fold!
kf   = KFold(n_splits=5, shuffle=True, random_state=0)

# one line: k scores, then mean & std
scores = cross_val_score(pipe, X, y, cv=kf, scoring="neg_mean_squared_error")
print(-scores.mean(), "+/-", scores.std())

# tune a hyperparameter with CV baked in
grid = GridSearchCV(pipe, {"ridge__alpha": [0.01, 0.1, 1, 10]}, cv=kf)
grid.fit(X, y); print(grid.best_params_)

A single train/test split can be lucky or unlucky. k-fold cross-validation rotates every chunk through validation, giving you k scores whose mean ± std is a trustworthy estimate — and the honest way to tune hyperparameters, as long as you fit preprocessing inside each fold and never touch the final test set.
