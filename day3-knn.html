🔨 Build it yourself

    
No training loop at all. Just measure distances and vote. ~25 lines of real logic.

    
    
      
  1. 1
    Euclidean distance between two points
    function dist(a, b) {
  let sum = 0;
  for (let i = 0; i < a.length; i++) sum += (a[i] - b[i]) ** 2;
  return Math.sqrt(sum);
}
// dist([1,2], [4,6]) → 5
    2. 
      
  2. 2
    Sort all training points by distance to the query
    function kNearest(data, query, k) {
  return data
    .map(([x, label]) => ({ label, d: dist(x, query) }))
    .sort((a, b) => a.d - b.d)
    .slice(0, k);
}
    3. 
      
  3. 3
    Majority vote among the k neighbors
    function vote(neighbors) {
  const counts = {};
  for (const { label } of neighbors) counts[label] = (counts[label] || 0) + 1;
  return Object.entries(counts).sort((a, b) => b[1] - a[1])[0][0];
}
    4. 
      
  4. 4
    Classify a new point — the full pipeline
    function classify(data, query, k) {
  const neighbors = kNearest(data, query, k);
  return vote(neighbors);
}
