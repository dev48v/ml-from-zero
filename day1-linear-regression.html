const predict = x => m * x + b;             // ŷ = m·x + b

    3
    One gradient-descent step
    
        
    

    function step(lr) {
  let dm = 0, db = 0;
  for (const [x, y] of data) {
    const err = predict(x) - y;     // how wrong, and which way
    dm += 2 * err * x;              // ∂loss/∂m
    db += 2 * err;                  // ∂loss/∂b
  }
  dm /= data.length; db /= data.length;
  m -= lr * dm;                     // step downhill
  b -= lr * db;
}
