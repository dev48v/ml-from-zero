    Measure how "mixed" a group is (Gini)
    function gini(pts){
  const p1 = pts.filter(p => p.label===1).length / pts.length;
  return 1 - (p1*p1 + (1-p1)*(1-p1));   // 0 = pure, 0.5 = 50/50
}
