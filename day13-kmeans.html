🎯 K-Means — Finding Groups With No Labels

K-Means alternates two moves: assign every point to its nearest centre, then move each centre to the average of its points. Repeat until nothing changes — unsupervised grouping with no labels at all.

    // each point is [x1, x2]. There is NO target column here —
// clustering is unsupervised, so the model never sees an answer.
const data = [
  [0.21, 0.78], [0.18, 0.71], [0.25, 0.83],   // (a natural blob)
  [0.80, 0.30], [0.84, 0.22], [0.76, 0.35],   // (another blob)
  // ...
];

  2. 
        
    2
    Pick K starting centres (k-means++)
    
        
    

    const K = 3;
const dist2 = (a, b) => (a[0]-b[0])**2 + (a[1]-b[1])**2;

// k-means++ : 1st centre random, each next centre is chosen far
// from the ones already picked → much better than pure random.
function initCentroids() {
  const centroids = [ data[Math.floor(Math.random()*data.length)] ];
  while (centroids.length < K) {
    const d = data.map(p => Math.min(...centroids.map(c => dist2(p, c))));
    const total = d.reduce((s, v) => s + v, 0);
    let r = Math.random() * total, i = 0;
    while ((r -= d[i]) > 0) i++;          // sample ∝ distance²
    centroids.push(data[i]);
  }
  return centroids.map(c => [...c]);
}

  3. 
        
    3
    Assign: each point → its nearest centre
    
        
    

    // for every point, find which centroid is closest.
function assign(centroids) {
  return data.map(p => {
    let best = 0, bestD = Infinity;
    centroids.forEach((c, k) => {
      const d = dist2(p, c);
      if (d < bestD) { bestD = d; best = k; }
    });
    return best;            // the cluster id this point belongs to
  });
}
    
          
        
    
      
    4. 
      
  4. 
        
    4
    Update: move each centre to its mean
    
        
    

    // recompute every centroid as the average of its assigned points.
function update(labels) {
  const sum = Array.from({length: K}, () => [0, 0, 0]); // x, y, count
  data.forEach((p, i) => {
    const k = labels[i];
    sum[k][0] += p[0]; sum[k][1] += p[1]; sum[k][2]++;
  });
  return sum.map(([sx, sy, n]) =>
    n ? [sx / n, sy / n] : [Math.random(), Math.random()]); // empty → re-seed
}

    5
    Loop until the centres stop moving
    
        
    

    let centroids = initCentroids(), labels, prev;
for (let it = 0; it < 100; it++) {
  labels = assign(centroids);            // step 3
  const next = update(labels);           // step 4
  // converged when no centroid moved more than a tiny epsilon
  const moved = next.some((c, k) => dist2(c, centroids[k]) > 1e-9);
  centroids = next;
  if (!moved) break;                     // done — it always stops
}

    Inertia: measure how tight the clusters are
    
        
    

    // inertia = total squared distance from each point to its centre.
// LOWER is tighter. It can only go down each iteration → convergence.
function inertia(centroids, labels) {
  return data.reduce((s, p, i) => s + dist2(p, centroids[labels[i]]), 0);
}
// run K-Means a few times with different seeds and keep the
// result with the smallest inertia — that beats one unlucky start.
